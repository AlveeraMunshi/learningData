Cheat Sheet of Python Sytnax

I prompted ChatGPT like this:

Can you give me a cheat sheet of Python syntax?
And it produced this:

Variables and Data Types 
# Variables
x = 5
y = "Hello, World!"

# Data Types
integer = 10
floating = 10.5
string = "Python"
boolean = True
list = [1, 2, 3]
tuple = (1, 2, 3)
dictionary = {"name": "John", "age": 30}
set_var = {1, 2, 3}
none_var = None
Arithmetic Operators 
a = 10
b = 3

addition = a + b       # 13
subtraction = a - b    # 7
multiplication = a * b # 30
division = a / b       # 3.3333
floor_division = a // b # 3
modulus = a % b        # 1
exponentiation = a ** b # 1000
Comparison Operators 
x == y # Equal to
x != y # Not equal to
x > y  # Greater than
x < y  # Less than
x >= y # Greater than or equal to
x <= y # Less than or equal to
Logical Operators 
and_operator = (x > 5 and y < 10)
or_operator = (x > 5 or y < 10)
not_operator = not(x > 5)
Control Flow - Conditional Statements 
if x > y:
    print("x is greater than y")
elif x == y:
    print("x is equal to y")
else:
    print("x is less than y")
Control Flow - Loops 
# For Loop
for i in range(5):
    print(i)

# While Loop
i = 0
while i < 5:
    print(i)
    i += 1
Functions 
def greet(name):
    return "Hello, " + name

# Calling the function
print(greet("Alice"))
Data Structures: Lists 
# Creating a list
fruits = ["apple", "banana", "cherry"]

# Accessing elements
print(fruits[0])  # apple

# Adding elements
fruits.append("orange")

# Removing elements
fruits.remove("banana")

# Slicing
print(fruits[1:3]) # ['cherry', 'orange']
Data Structures: Dictionaries 
# Creating a dictionary
person = {"name": "Alice", "age": 25}

# Accessing values
print(person["name"])  # Alice

# Adding key-value pairs
person["city"] = "New York"

# Removing key-value pairs
del person["age"]
Strings 
text = "Hello, World!"

# String concatenation
greeting = "Hello" + " " + "World"

# String formatting
formatted = f"{greeting}, it's {2024}!"

# Methods
print(text.lower()) # hello, world!
print(text.upper()) # HELLO, WORLD!
print(text.split(',')) # ['Hello', ' World!']
Exception Handling 
try:
    result = 10 / 0
except ZeroDivisionError:
    print("Cannot divide by zero")
finally:
    print("This will always execute")
File Handling 
# Writing to a file
with open('file.txt', 'w') as file:
    file.write("Hello, World!")

# Reading from a file
with open('file.txt', 'r') as file:
    content = file.read()
    print(content)
Modules and Imports 
# Importing a module
import math

# Using a module
print(math.sqrt(16))  # 4.0

# Importing specific functions
from math import pi, sqrt
print(pi)  # 3.14159
print(sqrt(16))  # 4.0
List Comprehensions 
# List comprehension
squares = [x**2 for x in range(10)]
Classes and Objects 
class Person:
    def __init__(self, name, age):
        self.name = name
        self.age = age

    def greet(self):
        return f"Hello, my name is {self.name}"

# Creating an object
person1 = Person("Alice", 25)
print(person1.greet())

Note: Defining classes is not covered in this class, so you can ignore that last part.
